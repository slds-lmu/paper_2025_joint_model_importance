% -----------------------------------------------------------------
\begin{frame}{Beyond the single best model}
  \begin{itemize}
    \item In tabular ML, many models reach indistinguishable predictive performance.
    \item For a tolerance $\varepsilon > 0$, the \emph{Rashomon set}
      \[
        \Rset \;=\; \bigl\{\, \fhDlam \,:\, \risk(\fhDlam) \le \risk(\fhDlams) \cdot (1+\varepsilon),\ \lamv \in \Lam \,\bigr\}
      \]
      collects every model within $\varepsilon$ of the best.\\[0.2em]
      {\footnotesize Notation: $\fhDlam$ = model trained on data $\D$ with hyperparameter configuration $\lamv \in \Lam$; $\lamv^{\ast}$ = risk-minimizing configuration; $\risk(\cdot)$ = risk.}
    \item Members of $\Rset$ can disagree substantially on individual predictions and on feature importance.
    \item \textbf{Question.} How are these equally good models related to one another, and what does the collective tell us beyond the single best?
  \end{itemize}
\end{frame}

% -----------------------------------------------------------------
\begin{frame}{Pipeline}
  \begin{enumerate}
    \item For every $\fhDlam \in \Rset$, compute the prediction vector $\hat{\yv}$ on a held-out 2/3 validation split.
    \item Build the pairwise \emph{behavioural distance} matrix $\mathbf{D}$ with entries
      \[
        d_{ij} \;=\; \|\hat{\yv}_i - \hat{\yv}_j\|_p \;=\; \Bigl(\sum_{o=1}^{n_{\text{val}}} \bigl|\hat y_{i,o} - \hat y_{j,o}\bigr|^p\Bigr)^{1/p},
      \]
      where $i, j$ index models in $\Rset$ and $o$ runs over the $n_{\text{val}}$ validation observations.
    \item View $\mathbf{D}$ three ways:
      \begin{itemize}
        \item hierarchical clustering (\texttt{hclust})
        \item classical MDS (\texttt{cmdscale}, embedding dimension $k = 2$)
        \item partitioning around medoids (\texttt{cluster::pam})
      \end{itemize}
  \end{enumerate}
  \vspace{0.5ex}
  {\footnotesize Distances computed for both Euclidean ($p = 2$) and Manhattan ($p = 1$); results shown are Euclidean (Manhattan tells the same story).}
\end{frame}

\begin{frame}{Scope --- behaviour is not importance}
  \begin{itemize}
    \item \textbf{What we measure:} behavioural distance
      \(
        d_{ij}^{\text{beh}} = \|\hat{\yv}_i - \hat{\yv}_j\|_p
      \)
      from validation predictions.
    \item \textbf{What ``joint model importance'' asks:} how do per-model importance vectors $\bm{\phi}^{(i)}$ vary across $i \in \Rset$? (Variable-Importance Cloud, VIC.)
    \item \textbf{Independent in principle:} identical predictions need not imply identical importance (and vice versa).
    \item \textbf{Compared empirically next slide:} same models, same colouring, behavioural distance left, VIC distance right.
  \end{itemize}
\end{frame}

\begin{frame}{Behavioural vs VIC --- what we see}
  \begin{itemize}
    \item \texttt{gc}: family separation largely preserved in both views, though a handful of \texttt{xgb} points overlap the \texttt{svm} cluster in VIC.
    \item \texttt{cs}: family separation preserved, but the within-family behavioural bimodality of trees (Slide~9) \textbf{vanishes} on the importance side.
    \item \texttt{bs}: single family in $\Rset$ --- uninformative.
    \item \texttt{st}: behavioural family separation \textbf{collapses} in VIC --- \texttt{nnet} and \texttt{svm} share the same importance profile despite predicting very differently.
    \item \textbf{Take-away:} behavioural clustering is \emph{not} a reliable proxy for importance clustering. The disagreement is empirical and \emph{task-dependent}.
  \end{itemize}
\end{frame}

% -----------------------------------------------------------------
\begin{frame}{First interpretation}
  \begin{itemize}
    \item Within $\Rset$, \textbf{behaviour has two-level structure:} learner family + position on the family's behavioural manifold.
    \item Across every task observed, family choice dominates behavioural variance --- not hyperparameter setting.
    \item The within-family manifold can be smooth and HP-aligned (\texttt{st}/\texttt{nnet} $\sim$ \texttt{decay}) or discrete and HP-orthogonal (\texttt{cs}/\texttt{tree}; \emph{causal mechanism still speculative}).
    \item \textbf{But behaviour is not importance:} family separation only partly survives in VIC (\texttt{gc}, \texttt{cs}), and the within-family discreteness of \texttt{cs}/\texttt{tree} vanishes entirely. For \texttt{st}, the families themselves collapse. \emph{Joint model importance therefore requires both views.}
  \end{itemize}
\end{frame}

% -----------------------------------------------------------------
\begin{frame}{Next steps}
  \begin{itemize}
    \item \textbf{$\varepsilon$ sensitivity.} How do behavioural and VIC clusters shift as the Rashomon tolerance opens?
    \item \textbf{Per-feature analysis.} Which features drive the behaviour-vs-VIC disagreement? Especially for the \texttt{st} collapse.
    \item \textbf{Medoids as representatives.} Treat the silhouette-tuned PAM medoids as a compact reduction of $\Rset$ for downstream interpretation.
    \item \textbf{Disagreement map.} Where in feature space do Rashomon-set models disagree the most?
    \item \textbf{Discrete tree mode on \texttt{cs}.} Is the behavioural bimodality driven by structural differences (root split, depth) that hyperparameters do not capture?
  \end{itemize}
\end{frame}

\begin{frame}{Backup --- the behavioural distance matrix in detail}
  Stack the Rashomon-set prediction vectors row-wise into a prediction matrix:
  \[
    \mathbf{P} \;=\;
    \begin{pmatrix}
      \hat y_{1,1} & \hat y_{1,2} & \cdots & \hat y_{1,n_{\text{val}}} \\
      \hat y_{2,1} & \hat y_{2,2} & \cdots & \hat y_{2,n_{\text{val}}} \\
      \vdots       & \vdots       &        & \vdots                    \\
      \hat y_{n,1} & \hat y_{n,2} & \cdots & \hat y_{n,n_{\text{val}}}
    \end{pmatrix}
    \;\in\; \R^{n \times n_{\text{val}}}
  \]
  with one row per model in $\Rset$ (here $n = |\Rset|$) and one column per held-out validation observation (here $n_{\text{val}}$).

  \vspace{0.5ex}
  Row-wise $L_p$ distances collapse the observation axis and yield the symmetric matrix
  \[
    \mathbf{D} \in \R^{n \times n}, \qquad d_{ij} \;=\; \|\hat{\yv}_i - \hat{\yv}_j\|_p.
  \]
  {\footnotesize Rows / columns of $\mathbf{D}$ now index \emph{models}, not observations. Hence \textbf{one MDS point = one model}; the cloud sits in $\R^n$ and is visualised in $\R^2$ via \texttt{cmdscale}.}
\end{frame}

% -----------------------------------------------------------------
\begin{frame}{Backup --- the three views of $\mathbf{D}$}
  \begin{itemize}
    \item \textbf{Hierarchical clustering} (\texttt{hclust}): bottom-up greedy merging of the two closest clusters; output is a dendrogram. Cluster count is fixed by where one cuts.
    \item \textbf{Classical MDS} (\texttt{cmdscale}, $k = 2$): eigendecomposition of the double-centred squared distance matrix, top-$k$ eigenvectors as coordinates. Each model becomes a point in $\R^k$ whose pairwise Euclidean distances approximate $\mathbf{D}$. For Euclidean $\mathbf{D}$ \textbf{equivalent to PCA} up to sign and rotation.
    \item \textbf{Partitioning around medoids} (\texttt{cluster::pam}): pick $k$ medoids (representative models) minimising the total distance from each model to its nearest medoid. Robust analogue of $k$-means for arbitrary distance matrices; best $k$ via average silhouette width.
  \end{itemize}
\end{frame}
